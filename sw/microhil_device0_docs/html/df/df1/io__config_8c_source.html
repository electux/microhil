181uint8_t microhil_module_init()
182{
183 stdio_init_all();
184
185 EPD_RST_PIN = 12;
186 EPD_DC_PIN = 8;
187 EPD_BL_PIN = 6;
188 EPD_CS_PIN = 9;
189 EPD_CLK_PIN = 10;
190 EPD_MOSI_PIN = 11;
191 EPD_SCL_PIN = 7;
192 EPD_SDA_PIN = 6;
193
194 spi_init(SPI_PORT, 1000 * 1000);
195 gpio_set_function(EPD_CLK_PIN, GPIO_FUNC_SPI);
196 gpio_set_function(EPD_MOSI_PIN, GPIO_FUNC_SPI);
197 microhil_gpio_init();
198 gpio_set_function(EPD_BL_PIN, GPIO_FUNC_PWM);
199
200 slice_num = pwm_gpio_to_slice_num(EPD_BL_PIN);
201
202 pwm_set_wrap(slice_num, 100);
203 pwm_set_chan_level(slice_num, PWM_CHAN_B, 1);
204 pwm_set_clkdiv(slice_num, 50);
205 pwm_set_enabled(slice_num, true);
206 i2c_init(i2c1, 100 * 1000);
207 gpio_set_function(EPD_SDA_PIN, GPIO_FUNC_I2C);
208 gpio_set_function(EPD_SCL_PIN, GPIO_FUNC_I2C);
209 gpio_pull_up(EPD_SDA_PIN);
210 gpio_pull_up(EPD_SCL_PIN);
211 printf("microhil_module_init OK \r\n");
212
213 return 0;
214}
