199bool microhil_module_init()
200{
201 bool status = false;
202
203 status = stdio_init_all();
204
205 if (!status)
206 {
209 return status;
210 }
211
212 epd_rst_pin = 12;
213 epd_dc_pin = 8;
214 epd_bl_pin = 6;
215 epd_cs_pin = 9;
216 epd_clk_pin = 10;
217 epd_mosi_pin = 11;
218 epd_scl_pin = 7;
219 epd_sda_pin = 6;
220
221 spi_init(MICROHIL_SPI_PORT_1, 1000 * 1000);
222 gpio_set_function(epd_clk_pin, GPIO_FUNC_SPI);
223 gpio_set_function(epd_mosi_pin, GPIO_FUNC_SPI);
224
225 microhil_gpio_init();
226
227 gpio_set_function(epd_bl_pin, GPIO_FUNC_PWM);
228 slice_num = pwm_gpio_to_slice_num(epd_bl_pin);
229 pwm_set_wrap(slice_num, 100);
230 pwm_set_chan_level(slice_num, PWM_CHAN_B, 1);
231 pwm_set_clkdiv(slice_num, 50);
232 pwm_set_enabled(slice_num, true);
233
234 i2c_init(MICROHIL_I2C_PORT_1, 100 * 1000);
235 gpio_set_function(epd_sda_pin, GPIO_FUNC_I2C);
236 gpio_set_function(epd_scl_pin, GPIO_FUNC_I2C);
237 gpio_pull_up(epd_sda_pin);
238 gpio_pull_up(epd_scl_pin);
239
240 printf("%s %s\n", MICROHIL_IO_CONFIG_INFO, MICROHIL_MODULE_INIT_OK);
241
242 return status;
243}
