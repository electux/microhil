73static inline pio_sm_config ws2812_program_get_default_config(uint offset)
74{
75 pio_sm_config config = pio_get_default_sm_config();
76 uint wrap_target = offset + WS2812_WRAP_TARGET;
77 uint wrap = offset + WS2812_WRAP;

102static inline void ws2812_program_init(ws2812_init* init)
103{
104 pio_gpio_init(init->pio, init->pin);
105 pio_sm_set_consecutive_pindirs(init->pio, init->sm, init->pin, 1, true);
106
107 pio_sm_config config = ws2812_program_get_default_config(init->offset);
108
109 sm_config_set_sideset_pins(&config, init->pin);
110 sm_config_set_out_shift(&config, false, true, init->rgbw ? 32 : 24);
111 sm_config_set_fifo_join(&config, PIO_FIFO_JOIN_TX);
112
113 int cycles_per_bit = (WS2812_T1 + WS2812_T2 + WS2812_T3);
114 float div = clock_get_hz(clk_sys) / (init->freq * cycles_per_bit);
115
116 sm_config_set_clkdiv(&config, div);
117 pio_sm_init(init->pio, init->sm, init->offset, &config);
118 pio_sm_set_enabled(init->pio, init->sm, true);
119}

199static inline void ws2812_parallel_program_init(ws2812_parallel_init* init)
200{
201 uint max_pin_count = init->pin_base + init->pin_count;
202
203 for (uint i = init->pin_base; i < max_pin_count; i++)
204 {
205 pio_gpio_init(init->pio, i);
206 }
207
208 pio_sm_set_consecutive_pindirs(
209 init->pio, init->sm, init->pin_base, init->pin_count, true
210 );
211
212 pio_sm_config config = ws2812_parallel_program_get_default_config(
213 init->offset
214 );
215
216 sm_config_set_out_shift(&config, true, true, 32);
217 sm_config_set_out_pins(&config, init->pin_base, init->pin_count);
218 sm_config_set_set_pins(&config, init->pin_base, init->pin_count);
219 sm_config_set_fifo_join(&config, PIO_FIFO_JOIN_TX);
220
221 int cycles_per_bit = (
222 WS2812_PARALLEL_T1 + WS2812_PARALLEL_T2 + WS2812_PARALLEL_T3
223 );
224
225 float div = clock_get_hz(clk_sys) / (init->freq * cycles_per_bit);
226
227 sm_config_set_clkdiv(&config, div);
228 pio_sm_init(init->pio, init->sm, init->offset, &config);
229 pio_sm_set_enabled(init->pio, init->sm, true);
230}
