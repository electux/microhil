71void microhil_channel_switch(int id)
72{
73 unsigned int gpio = id_to_gpio(id);
74
75 switch (gpio)
76 {
77 case MICROHIL_ALL_CHANNELS_OFF:
78 microhil_all_channels_off();
79 break;
80 case MICROHIL_ALL_CHANNELS_ON:
81 microhil_all_channels_on();
82 break;
83 case MICROHIL_CHANNELS_ERROR:
84 printf("****Error*****\r\n\n");
85 break;
86 default:
87 // TODO check gpio value
88 gpio_put(gpio, channels[id - 1]);
89
90 if (channels[id - 1] == 1)
91 {
92 put_pixel(urgb_u32(255, 0, 0));
93 microhil_write_pwm(80); // buzzer
94 microhil_delay_ms(100);
95 microhil_write_pwm(0);
96 printf("****Relay 1 on*****\r\n\n");
97 }
98 else if (channels[id - 1] == 0)
99 {
100 printf("****Relay 1 off*****\r\n\n");
101 }
102
103 channels[id - 1] = !channels[id - 1];
104 break;
105 }
106}
