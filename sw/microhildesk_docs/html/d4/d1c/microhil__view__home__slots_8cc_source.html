21void MHViewHome::onChannelChanged(Channel id)
22{
23 auto ix = channelToInt(id);
24 bool status = m_enableChannels[ix]->get_active();
25
26 if (status)
27 {
28 enableChannel(id);
29 }
30 else
31 {
32 disableChannel(id);
33 }
34
35 m_channelEnabled.emit(id, status);
36}

38void MHViewHome::onTypeSelected(Channel id)
39{
40 auto ix = channelToInt(id);
41
42 if (m_enableChannels[ix]->get_active())
43 {
44 int controlType = m_selectControlChannels[ix]->get_active_row_number();
45
46 if (controlTypeToInt(channelControlType::MICROHIL_TOGGLE_BUTTON) == 0)
47 {
48 toggleModeChannel(id);
49 }
50
51 if (controlTypeToInt(channelControlType::MICROHIL_TIMER_BUTTON) == 1)
52 {
53 timerModeChannel(id);
54 }
55
56 m_channelControlType.emit(id, controlType);
57 }
58}
