51void MHViewSerial::onOkSerialTriggered()
52{

54
55 MHVecUInt serialParams{};
56 serialParams.push_back(m_baudRate->get_active_row_number());
57 serialParams.push_back(m_dataBits->get_active_row_number());
58 serialParams.push_back(m_parity->get_active_row_number());
59 serialParams.push_back(m_stopBits->get_active_row_number());
60
63 m_serialSetup.emit(devicePath, serialParams);
64
65 this->hide();
