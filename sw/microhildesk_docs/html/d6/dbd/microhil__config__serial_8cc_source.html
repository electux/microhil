146bool MHConfig::validateSerialSettings()
147{
150 const auto configCheck =
151 (m_config.has_key(kConfigSerial, kConfigDevice) &&
152 m_config.has_key(kConfigSerial, kConfigBaudRate) &&
153 m_config.has_key(kConfigSerial, kConfigDataBits) &&
154 m_config.has_key(kConfigSerial, kConfigParity) &&
155 m_config.has_key(kConfigSerial, kConfigStopBits));
156
157 if (!configCheck)
158 {
161 return false;
162 }
163
164 return true;
165}
