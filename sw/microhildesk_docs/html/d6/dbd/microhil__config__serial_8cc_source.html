140bool MHConfig::validateSerialSettings()
141{
144 const auto configCheck = (
145 m_config.has_key(kConfigSerialSection, kConfigSerialDevice) &&
146 m_config.has_key(kConfigSerialSection, kConfigSerialBaudRate) &&
147 m_config.has_key(kConfigSerialSection, kConfigSerialDataBits) &&
148 m_config.has_key(kConfigSerialSection, kConfigSerialParity) &&
149 m_config.has_key(kConfigSerialSection, kConfigSerialStopBits)
150 );
151
152 if (!configCheck)
153 {
156 return false;
157 }
158
159 return true;
160}
