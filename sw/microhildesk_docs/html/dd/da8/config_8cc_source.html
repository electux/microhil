45bool Config::load()
46{
47 std::ifstream file(m_fileName);
48
49 std::cout << "Load configuration..." << std::endl;
50
51 if (!file.is_open())
52 {
53 std::cerr << "Unable to open file: " << m_fileName << std::endl;
54 return false;
55 }
56
57 std::string line;
58 std::unordered_map<std::string, std::string> configMap;
59
60 while (std::getline(file, line))
61 {
62 std::istringstream iss(line);
63 std::string key, value;
64
65 if (std::getline(iss, key, '=') && std::getline(iss, value))
66 {
67 if (!m_serialConfig.validateKey(key) && !m_logConfig.validateKey(key))
68 {
69 std::cerr << "Invalid configuration key: " << key << std::endl;
70 return false;
71 }
72 configMap[key] = value;
73 }
74 }
75
76 file.close();
77
80 std::initializer_list<ModelSerial::ModelSerialKey> serialKeys = {
81 ModelSerial::ModelSerialKey::Device,
82 ModelSerial::ModelSerialKey::Baud,
83 ModelSerial::ModelSerialKey::Data,
84 ModelSerial::ModelSerialKey::Parity,
85 ModelSerial::ModelSerialKey::Stop
86 };
87 for (const auto key : serialKeys)
88 {
89 auto serialKey = m_serialConfig.toString(key);
90 m_serialConfig.add(serialKey, configMap[serialKey]);
91 }
92
95 auto logLevelKey = m_logConfig.toString(ModelLog::ModelLogKey::LogLevel);
96 m_logConfig.add(logLevelKey, configMap[logLevelKey]);
97
98 std::cout << "Load configuration done." << std::endl;
99
100 return true;
101}

122 ModelSerial::ModelSerialKey::Stop
123 };
124
125 for (const auto key : serialKeys)
126 {
127 auto serialKey = m_serialConfig.toString(key);
128 file << serialKey << "=" << m_serialConfig.getEntity(serialKey) << std::endl;

130
133 auto logLevelKey = m_logConfig.toString(ModelLog::ModelLogKey::LogLevel);
134 file << logLevelKey << "=" << m_logConfig.getEntity(logLevelKey) << std::endl;

139 return true;
