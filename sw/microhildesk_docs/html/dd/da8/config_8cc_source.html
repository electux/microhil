53bool Config::load()
54{
55 std::ifstream file(m_fileName);
56 std::cout << "Load configuration..." << std::endl;
57
58 if (!file.is_open())
59 {
60 std::cerr << "Unable to open file: " << m_fileName << std::endl;
61 return false;
62 }
63
64 std::string line;
65 while (std::getline(file, line))
66 {
67 if (line.empty() || line[0] == '#')
68 {
69 continue;
70 }
71
72 std::istringstream iss(line);
73 std::string key, value;
74
75 if (std::getline(iss, key, '=') && std::getline(iss, value))
76 {
77 if (m_serialConfig.validateKey(key))
78 {
79 m_serialConfig.add(key, value);
80 }
81 else if (m_logConfig.validateKey(key))
82 {
83 m_logConfig.add(key, value);
84 }
85 else
86 {
87 std::cerr << "Warning: Unknown configuration key: " << key << std::endl;
88 return false;
89 }
90 }
91 }
92
93 file.close();
94 std::cout << "Load configuration done." << std::endl;
95 return true;
96}

98bool Config::store()
99{
100 std::ofstream file(m_fileName);
101 std::cout << "Store configuration..." << std::endl;
102
103 if (!file.is_open())
104 {
105 std::cerr << "Unable to open file: " << m_fileName << std::endl;
106 return false;
107 }
108
109 auto writeModel = [&](const auto& model)
110 {
111 for (const auto& [key, value] : model.getAllEntries())
112 {
113 file << key << "=" << value << "\n";
114 }
115 };
116
117 writeModel(m_serialConfig);
118 writeModel(m_logConfig);
119
120 file.close();
121 std::cout << "Store configuration done." << std::endl;
122 return true;
123}
