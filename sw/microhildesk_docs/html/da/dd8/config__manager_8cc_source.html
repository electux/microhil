43ConfigManager::ConfigManager()
44{
45 m_fileName = Glib::build_filename(Glib::get_home_dir(), configFile.data());
46 auto dirPath = Glib::path_get_dirname(m_fileName);
47
48 try
49 {
50 if (!std::filesystem::exists(dirPath))
51 {
52 std::filesystem::create_directories(dirPath);
53 }
54
55 if (!std::filesystem::exists(m_fileName))
56 {
57 std::ofstream file(m_fileName);
58 if (!file)
59 {
60 throw std::ios_base::failure("Unable to create file: " + m_fileName);
61 }
62 defaultConfigStore();
63 }
64
65 std::cout << "Setup configuration file path: " << m_fileName << std::endl;
66 load();
67 } catch (const std::exception& e)
68 {
69 std::cerr << "Error during ConfigManager initialization: " << e.what() << std::endl;
70 }
71}

93bool ConfigManager::load()
94{
95 std::ifstream file(m_fileName);
96 std::cout << "Loading configuration..." << std::endl;
97
98 if (!file.is_open())
99 {
100 std::cerr << "Unable to open file: " << m_fileName << std::endl;
101 return false;
102 }
103
104 std::string line;
105 while (std::getline(file, line))
106 {
107 if (line.empty() || line[0] == configCommentChar)
108 {
109 continue;
110 }
111
112 std::istringstream iss(line);
113 std::string key, value;
114
115 if (std::getline(iss, key, configAssignDelimiter) && std::getline(iss, value))
116 {
117 key.erase(0, key.find_first_not_of(" \t\n\r"));
118 key.erase(key.find_last_not_of(" \t\n\r") + 1);
119 value.erase(0, value.find_first_not_of(" \t\n\r"));
120 value.erase(value.find_last_not_of(" \t\n\r") + 1);
121
122 if (m_serialConfig.validateKey(key))
123 {
124 m_serialConfig.add(key, value);
125 }
126 else if (m_logConfig.validateKey(key))
127 {
128 m_logConfig.add(key, value);
129 }
130 else
131 {
132 std::cerr << "Warning: Unknown configuration key: " << key << std::endl;
133 return false;
134 }
135 }
136 }
137
138 file.close();
139 std::cout << "Load configuration done." << std::endl;
140 return true;
141}

143bool ConfigManager::store()
144{
145 std::ofstream file(m_fileName);
146 std::cout << "Storing configuration..." << std::endl;
147
148 if (!file.is_open())
149 {
150 std::cerr << "Unable to open file: " << m_fileName << std::endl;
151 return false;
152 }
153
154 auto writeModel = [&](const auto& model)
155 {
156 for (const auto& [key, value] : model.getAllEntries())
157 {
158 file << key << configAssignDelimiter << value << "\n";
159 }
160 };
161
162 writeModel(m_serialConfig);
163 writeModel(m_logConfig);
164
165 file.close();
166 std::cout << "Store configuration done." << std::endl;
167 return true;
168}
