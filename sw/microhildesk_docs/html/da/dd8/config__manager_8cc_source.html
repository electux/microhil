69ConfigManager::ConfigManager()
70{
71 m_fileName = Glib::build_filename(Glib::get_home_dir(), cConfigFile.data());
72 auto dirPath = Glib::path_get_dirname(m_fileName);
73
74 try
75 {
76 if (!std::filesystem::exists(dirPath))
77 {
78 std::filesystem::create_directories(dirPath);
79 }
80
81 if (!std::filesystem::exists(m_fileName))
82 {
83 std::ofstream file(m_fileName);
84 if (!file)
85 {
86 throw std::ios_base::failure("Unable to create file: " + m_fileName);
87 }
88 defaultConfigStore();
89 }
90
91 std::cout << "Setup configuration file path: " << m_fileName << std::endl;
92 load();
93 } catch (const std::exception& e)
94 {
95 std::cerr << "Error during ConfigManager initialization: " << e.what() << std::endl;
96 }
97}

129bool ConfigManager::load()
130{
131 std::ifstream file(m_fileName);
132 std::cout << "Loading configuration..." << std::endl;
133
134 if (!file.is_open())
135 {
136 std::cerr << "Unable to open file: " << m_fileName << std::endl;
137 return false;
138 }
139
140 std::string line;
141 while (std::getline(file, line))
142 {
143 if (line.empty() || line[0] == cConfigCommentChar)
144 {
145 continue;
146 }
147
148 std::istringstream iss(line);
149 std::string key, value;
150
151 if (std::getline(iss, key, cConfigAssignDelimiter) && std::getline(iss, value))
152 {
153 key.erase(0, key.find_first_not_of(" \t\n\r"));
154 key.erase(key.find_last_not_of(" \t\n\r") + 1);
155 value.erase(0, value.find_first_not_of(" \t\n\r"));
156 value.erase(value.find_last_not_of(" \t\n\r") + 1);
157
158 if (m_serialConfig.validateKey(key))
159 {
160 m_serialConfig.add(key, value);
161 }
162 else if (m_logConfig.validateKey(key))
163 {
164 m_logConfig.add(key, value);
165 }
166 else if (m_controlConfig.validateKey(key))
167 {
168 m_controlConfig.add(key, value);
169 }
170 else
171 {
172 std::cerr << "Warning: Unknown configuration key: " << key << std::endl;
173 return false;
174 }
175 }
176 }
177
178 file.close();
179 std::cout << "Load configuration done." << std::endl;
180 return true;
181}

183bool ConfigManager::store()
184{
185 std::ofstream file(m_fileName);
186 std::cout << "Storing configuration..." << std::endl;
187
188 if (!file.is_open())
189 {
190 std::cerr << "Unable to open file: " << m_fileName << std::endl;
191 return false;
192 }
193
194 auto writeModel = [&](const auto& model)
195 {
196 for (const auto& [key, value] : model.getAllEntries())
197 {
198 file << key << cConfigAssignDelimiter << value << "\n";
199 }
200 };
201
202 writeModel(m_controlConfig);
203 writeModel(m_serialConfig);
204 writeModel(m_logConfig);
205
206 file.close();
207 std::cout << "Store configuration done." << std::endl;
208 return true;
209}
