98bool ConfigManager::load()
99{
100 std::ifstream file(m_fileName);
101 std::cout << "Loading configuration..." << std::endl;
102
103 if (!file.is_open())
104 {
105 std::cerr << "Unable to open file: " << m_fileName << std::endl;
106 return false;
107 }
108
109 std::string line;
110 while (std::getline(file, line))
111 {
112 // Skip empty lines and comments
113 if (line.empty() || line[0] == cConfigCommentChar)
114 {
115 continue;
116 }
117
118 std::istringstream iss(line);
119 std::string key, value;
120
121 if (std::getline(iss, key, cConfigAssignDelimiter) && std::getline(iss, value))
122 {
123 // Trim whitespace
124 key.erase(0, key.find_first_not_of(" \t\n\r"));
125 key.erase(key.find_last_not_of(" \t\n\r") + 1);
126 value.erase(0, value.find_first_not_of(" \t\n\r"));
127 value.erase(value.find_last_not_of(" \t\n\r") + 1);
128
129 // Map keys to specific models
130 if (m_config.validateKey(key))
131 {
132 m_config.add(key, value);
133 }
134 else
135 {
136 std::cerr << "Warning: Unknown configuration key: " << key << std::endl;
137 continue;
138 }
139 }
140 }
141
142 file.close();
143 std::cout << "Load configuration done." << std::endl;
144 return true;
145}

151bool ConfigManager::store()
152{
153 std::ofstream file(m_fileName);
154 std::cout << "Storing configuration..." << std::endl;
155
156 if (!file.is_open())
157 {
158 std::cerr << "Unable to open file: " << m_fileName << std::endl;
159 return false;
160 }
161
162 // Internal helper lambda for writing model entries
163 auto writeModel = [&](const auto& model)
164 {
165 for (const auto& [key, value] : model.getAllEntries())
166 {
167 file << key << cConfigAssignDelimiter << value << "\n";
168 }
169 };
170
171 writeModel(m_config);
172
173 file.close();
174 std::cout << "Store configuration done." << std::endl;
175 return true;
176}
