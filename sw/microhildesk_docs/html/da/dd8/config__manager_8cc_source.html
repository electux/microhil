106bool ConfigManager::load()
107{
108 std::ifstream file(m_fileName);
109 std::cout << "Loading configuration..." << std::endl;
110
111 if (!file.is_open())
112 {
113 std::cerr << "Unable to open file: " << m_fileName << std::endl;
114 return false;
115 }
116
117 std::string line;
118 while (std::getline(file, line))
119 {
120 // Skip empty lines and comments
121 if (line.empty() || line[0] == cConfigCommentChar)
122 {
123 continue;
124 }
125
126 std::istringstream iss(line);
127 std::string key, value;
128
129 if (std::getline(iss, key, cConfigAssignDelimiter) && std::getline(iss, value))
130 {
131 // Trim whitespace
132 key.erase(0, key.find_first_not_of(" \t\n\r"));
133 key.erase(key.find_last_not_of(" \t\n\r") + 1);
134 value.erase(0, value.find_first_not_of(" \t\n\r"));
135 value.erase(value.find_last_not_of(" \t\n\r") + 1);
136
137 // Map keys to specific models
138 if (m_config->validateKey(key))
139 {
140 m_config->add(key, value);
141 }
142 else
143 {
144 std::cerr << "Warning: Unknown configuration key: " << key << std::endl;
145 continue;
146 }
147 }
148 }
149
150 file.close();
151 std::cout << "Load configuration done." << std::endl;
152 return true;
153}

159bool ConfigManager::store()
160{
161 std::ofstream file(m_fileName);
162 std::cout << "Storing configuration..." << std::endl;
163
164 if (!file.is_open())
165 {
166 std::cerr << "Unable to open file: " << m_fileName << std::endl;
167 return false;
168 }
169
170 // Internal helper lambda for writing model entries
171 auto writeModel = [&](const auto& model)
172 {
173 for (const auto& [key, value] : model.getAllEntries())
174 {
175 file << key << cConfigAssignDelimiter << value << "\n";
176 }
177 };
178
179 writeModel(*m_config);
180
181 file.close();
182 std::cout << "Store configuration done." << std::endl;
183 return true;
184}
