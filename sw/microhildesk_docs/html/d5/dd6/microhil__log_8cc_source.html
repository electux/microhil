122UString MHLog::getCurrentDateTime() const
123{
124 time_t rawTime;
125 char buffer[80];
126
129 time(&rawTime);
130 const struct tm *timeInfo = localtime(&rawTime);
131 strftime(buffer, 80, "%d-%m-%Y %I:%M:%S", timeInfo);
132
133 return UString(buffer);
