105MHString MHLog::getCurrentDateTime() const
106{

110 time(&rawTime);
111 const struct tm *timeInfo = localtime(&rawTime);
112 strftime(buffer, 80, "%d-%m-%Y %I:%M:%S", timeInfo);
113

117MHString MHLog::toStringLogType(const LogLevel level) const
