29void SerialCom::open()
30{
31 std::cout << "Open serial port..." << std::endl;
32
33 if (m_device.empty()) {
34 std::cerr << "Error: Device path is empty!" << std::endl;
35 return;
36 }
37
38 if (!m_serialPort.IsOpen())
39 {
40 try {
41 m_serialPort.Open(std::string(m_device));
42 std::cout << "Open serial port done." << std::endl;
43 } catch (const std::exception &e) {
44 std::cerr << "Error opening serial port: " << e.what() << std::endl;
45 }
46 }

60void SerialCom::setup(const SerialParams &params)
61{
62 std::cout << "Setup serial port..." << std::endl;
63
64 if (m_serialPort.IsOpen())
65 {
66 m_serialPort.Close();
67 }
68
71 m_device = params.device;
72
73 try {
74 open();
75
76 if (m_serialPort.IsOpen())
77 {
80 m_serialPort.SetBaudRate(params.baud);
81 m_serialPort.SetCharacterSize(params.data);
82 m_serialPort.SetParity(params.parity);
83 m_serialPort.SetStopBits(params.stop);
84 m_serialPort.SetFlowControl(params.flow);
85
86 std::cout << "Setup serial port done." << std::endl;
87 }
88 else
89 {
90 std::cerr << "Error: Port could not be opened, skipping configuration." << std::endl;
91 }
92 } catch (const std::exception &e) {
93 std::cerr << "Error setting up serial port: " << e.what() << std::endl;
94 }
95}
