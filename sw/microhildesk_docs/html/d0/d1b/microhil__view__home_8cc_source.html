125MHViewHome::MHViewHome(BaseObjectType *object, MHRPtr<Gtk::Builder> const &ui) : Gtk::ApplicationWindow(object), m_ui{ui}
126{
128 // Bind Connect menu-item
129 m_connect = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kConnectId));
130
132 // Bind Disconnect menu-item
133 m_disconnect = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kDisconnectId));
134
136 // Bind Quit menu-item
137 m_quit = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kQuitId));
138
140 // Bind Serial Settings menu-item
141 m_serialSettings = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kSerialSettingsId));
142
144 // Bind Log Settings menu-item
145 m_logSettings = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kLogSettingsId));
146
148 // Bind About menu-item
149 m_about = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kAboutId));
150
152 // Bind channels widgets
153 for (int i = 0; i < kNumberOfViewChannels; i++)
154 {
155 m_enableChannels.push_back(MHRPtr<Gtk::CheckButton>::cast_dynamic(m_ui->get_object(kCheckButtonsId[i])));
156 m_selectControlChannels.push_back(MHRPtr<Gtk::ComboBoxText>::cast_dynamic(m_ui->get_object(kComboBoxTextsId[i])));
157 m_toggleChannels.push_back(MHRPtr<Gtk::ToggleButton>::cast_dynamic(m_ui->get_object(kToggleButtonsId[i])));
158 m_spinTimerChannels.push_back(MHRPtr<Gtk::SpinButton>::cast_dynamic(m_ui->get_object(kTimerSpinButtonsId[i])));
159 m_toggleTimerChannels.push_back(MHRPtr<Gtk::ToggleButton>::cast_dynamic(m_ui->get_object(kTimerToggleButtonsId[i])));
160 m_stautsTimerChannels.push_back(MHRPtr<Gtk::ProgressBar>::cast_dynamic(m_ui->get_object(kTimerProgressBarStatus0Id[i])));
161 }
162
164 // Map channels (signals and slots)
165 mapping();
166
168 // Disabled all channels by default
169 for (int i = 0; i < kNumberOfViewChannels; i++)
170 {
171 disableChannel(static_cast<Channel>(i));
172 }
173}
