131MHViewHome::MHViewHome(BaseObjectType *object, MHRPtr<Gtk::Builder> const &ui)
132 : Gtk::ApplicationWindow(object), m_ui{ui}
133{
135 // Binds Connect menu-item
136 m_connect = MHRPtr<Gtk::MenuItem>::cast_dynamic(
137 m_ui->get_object(kConnectId)
138 );
139
141 // Binds Disconnect menu-item
142 m_disconnect = MHRPtr<Gtk::MenuItem>::cast_dynamic(
143 m_ui->get_object(kDisconnectId)
144 );
145
147 // Binds Quit menu-item
148 m_quit = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kQuitId));
149
151 // Binds Serial Settings menu-item
152 m_serialSettings = MHRPtr<Gtk::MenuItem>::cast_dynamic(
153 m_ui->get_object(kSerialSettingsId)
154 );
155
157 // Binds Log Settings menu-item
158 m_logSettings = MHRPtr<Gtk::MenuItem>::cast_dynamic(
159 m_ui->get_object(kLogSettingsId)
160 );
161
163 // Binds About menu-item
164 m_about = MHRPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kAboutId));
165
167 // Binds channels widgets
168 for (int i = 0; i < kNumberOfViewChannels; i++)
169 {
170 m_enableChannels.push_back(MHRPtr<Gtk::CheckButton>::cast_dynamic(
171 m_ui->get_object(kCheckButtonsId[i]))
172 );
173 m_selectControlChannels.push_back(
174 MHRPtr<Gtk::ComboBoxText>::cast_dynamic(
175 m_ui->get_object(kComboBoxTextsId[i])
176 )
177 );
178 m_toggleChannels.push_back(
179 MHRPtr<Gtk::ToggleButton>::cast_dynamic(
180 m_ui->get_object(kToggleButtonsId[i])
181 )
182 );
183 m_spinTimerChannels.push_back(
184 MHRPtr<Gtk::SpinButton>::cast_dynamic(
185 m_ui->get_object(kTimerSpinButtonsId[i])
186 )
187 );
188 m_toggleTimerChannels.push_back(
189 MHRPtr<Gtk::ToggleButton>::cast_dynamic(
190 m_ui->get_object(kTimerToggleButtonsId[i])
191 )
192 );
193 m_stautsTimerChannels.push_back(
194 MHRPtr<Gtk::ProgressBar>::cast_dynamic(
195 m_ui->get_object(kTimerProgressBarStatus0Id[i])
196 )
197 );
198 }
199
201 // Maps channels (signals and slots)
202 mapping();
203
205 // Disables all channels by default
206 for (int i = 0; i < kNumberOfViewChannels; i++)
207 {
208 disableChannel(static_cast<Channel>(i));
209 }
210
213 m_disconnect->set_sensitive(false);
214}
