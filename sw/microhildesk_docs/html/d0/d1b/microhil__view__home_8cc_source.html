94MHViewHome::MHViewHome(BaseObjectType *object, RPtr<Gtk::Builder> const &ui)
95 : Gtk::ApplicationWindow(object)
96 , m_ui{ui}
97{
99 // Binds Connect menu-item
100 m_connect = RPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kConnectId));
101
103 // Binds Disconnect menu-item
104 m_disconnect =
105 RPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kDisconnectId));
106
108 // Binds Quit menu-item
109 m_quit = RPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kQuitId));
110
112 // Binds Serial Settings menu-item
113 m_serialSettings =
114 RPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kSerialId));
115
117 // Binds Log Settings menu-item
118 m_logSettings = RPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kLogId));
119
121 // Binds About menu-item
122 m_about = RPtr<Gtk::MenuItem>::cast_dynamic(m_ui->get_object(kAboutId));
123
125 // Binds channels widgets
126 for (int i = 0; i < kNumberOfViewChannels; i++)
127 {
128 m_enableChannels.push_back(RPtr<Gtk::CheckButton>::cast_dynamic(
129 m_ui->get_object(kCheckButtonsId[i])
130 ));
131
132 m_selectControlChannels.push_back(RPtr<Gtk::ComboBoxText>::cast_dynamic(
133 m_ui->get_object(kComboBoxTextsId[i])
134 ));
135
136 m_toggleChannels.push_back(RPtr<Gtk::ToggleButton>::cast_dynamic(
137 m_ui->get_object(kToggleButtonsId[i])
138 ));
139
140 m_spinTimerChannels.push_back(RPtr<Gtk::SpinButton>::cast_dynamic(
141 m_ui->get_object(kTimerSpinButtonsId[i])
142 ));
143
144 m_toggleTimerChannels.push_back(RPtr<Gtk::ToggleButton>::cast_dynamic(
145 m_ui->get_object(kTimerToggleButtonsId[i])
146 ));
147
148 m_stautsTimerChannels.push_back(RPtr<Gtk::ProgressBar>::cast_dynamic(
149 m_ui->get_object(kTimerProgressBarStatus0Id[i])
150 ));
151 }
152
154 // Maps channels (signals and slots)
155 mapping();
156
158 // Disables all channels by default
159 for (int i = 0; i < kNumberOfViewChannels; i++)
160 {
161 disableChannel(static_cast<Channel>(i));
162 }
163
166 m_disconnect->set_sensitive(false);
167}
