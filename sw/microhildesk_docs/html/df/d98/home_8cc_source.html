61AppHome::AppHome()
62{
65 set_title(homeTitle.data());
66 set_default_size(homeWidth, homeHeight);
67 set_resizable(false);
68 set_show_menubar(true);
69 m_boxRoot.set_orientation(Gtk::Orientation::HORIZONTAL);
70 set_child(m_boxRoot);
71
72 for(unsigned int i = 0; i < numOfChannels; i++)
73 {
74 m_boxChannels.emplace_back(Gtk::Orientation::VERTICAL);
75 auto& currentBox = m_boxChannels.back();
76 m_enableChannels.emplace_back(std::format("{} {}", homeEnChannelLabel.data(), i));

78 m_selectControlChannels.emplace_back();

83 currentBox.append(m_selectControlChannels.back());
84 m_labelToggleChannels.emplace_back(std::format("{} {}", homeToggleChannelLabel.data(), i));
85 currentBox.append(m_labelToggleChannels.back());
86 m_toggleChannels.emplace_back();
87 m_toggleChannels.back().set_label(homeToggleChannelButtonActivate.data());
88 currentBox.append(m_toggleChannels.back());
89 m_labelTimerChannels.emplace_back(std::format("{} {}", homeTimerChannelLabel.data(), i));
90 currentBox.append(m_labelTimerChannels.back());
91 m_spinTimerChannels.emplace_back();
92 currentBox.append(m_spinTimerChannels.back());
93 m_toggleTimerChannels.emplace_back();
94 m_toggleTimerChannels.back().set_label(homeTimerChannelButtonStart.data());
95 currentBox.append(m_toggleTimerChannels.back());
96 m_statusTimerChannels.emplace_back();
97 m_statusTimerChannels.back().set_fraction(homeStatusFraction);
98 currentBox.append(m_statusTimerChannels.back());
99 currentBox.set_margin(homeBoxChannelMargin);
100 currentBox.set_spacing(homeBoxChannelSpacing);
101 m_boxRoot.append(currentBox);
102 mapping(i);
103 }
