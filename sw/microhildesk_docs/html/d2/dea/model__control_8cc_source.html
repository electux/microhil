36Entities ModelControl::getAllEntries() const
37{
38 static const std::vector<ModelControlKey> keys =
39 {
40 ModelControlKey::Enable,
41 ModelControlKey::Mode,
42 ModelControlKey::Toggle,
43 ModelControlKey::Timer,
44 ModelControlKey::TimerEnable
45 };
46
47 Entities entries;
48 for (const auto& key : keys)

50 std::string keyStr = toString(key);
51 entries[keyStr] = getEntity(keyStr);
52 }
53
54 return entries;

57std::string ModelControl::toString(const ModelControlKey &key) const
