37Entities ModelSerial::getAllEntries() const
38{
39 static const std::vector<ModelSerialKey> keys =
40 {
41 ModelSerialKey::Device,
42 ModelSerialKey::Baud,
43 ModelSerialKey::Data,
44 ModelSerialKey::Parity,
45 ModelSerialKey::Stop,
46 ModelSerialKey::Flow
47 };
48
49 Entities entries;
50 for (const auto& key : keys)
51 {
52 std::string_view keyStrView = toString(key);
53 std::string keyStr{keyStrView};
54 entries.emplace(std::move(keyStr), getEntity(keyStr));
55 }
56
57 return entries;
58}
