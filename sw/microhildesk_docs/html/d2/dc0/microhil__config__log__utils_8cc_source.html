48MHString MHConfig::logLevelToUnicodeString(const int logLevel)
49{
52 auto logLevelType = static_cast<LogLevelConfig>(logLevel);
53
54 switch (logLevelType)
55 {
56 case LogLevelConfig::MICROHIL_INFO_CONFIG:
57 return kConfigLogLevelInfo;
58 case LogLevelConfig::MICROHIL_WARNING_CONFIG:
59 return kConfigLogLevelWarning;
60 case LogLevelConfig::MICROHIL_ERROR_CONFIG:
61 return kConfigLogLevelError;
62 }
63
64 return kConfigLogLevelInfo;
65}
