159std::string AppHome::extract_param_value_by_index(const std::string_view &input, size_t index)
160{
161 auto words = input | std::views::split(cSpaceDelimiter) | std::views::transform(
162 [](auto &&rng)
163 {
164 return std::string_view(&*rng.begin(), std::ranges::distance(rng));
165 }
166 );
167
168 auto it = words.begin();
169 for (size_t i = 0; i < index && it != words.end(); ++i)
170 {
171 ++it;
172 }
173
174 return (it != words.end()) ? std::string(*it) : std::string(cEmptyString);
175}

177std::string AppHome::update_param_value_by_index(const std::string_view &input, size_t index, const std::string_view &newValue)
178{
179 auto view = input | std::views::split(cSpaceDelimiter) | std::views::transform(
180 [](auto &&range)
181 {
182 return std::string(range.begin(), range.end());
183 }
184 );
185
186 std::vector<std::string> words{view.begin(), view.end()};
187
188 if (index < words.size())
189 {
190 words[index] = newValue;
191 }
192
193 if (words.empty())
194 {
195 return std::string(cEmptyString);
196 }
197
198 return std::accumulate(
199 std::next(words.begin()),
200 words.end(),
201 words[0],
202 [](std::string a, std::string b)
203 {
204 return std::move(a) + std::string(cSpaceString) + b;
205 }
206 );
207}
