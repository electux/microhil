138std::string AppHome::extract_param_value_by_index(const std::string &input, size_t index)
139{
140 auto words = input | std::views::split(' ') | std::views::transform(
141 [](auto &&rng) {return std::string_view(&*rng.begin(), std::ranges::distance(rng));}
142 );
143
144 auto it = words.begin();
145 for (size_t i = 0; i < index && it != words.end(); ++i)
146 {
147 ++it;
148 }
149
150 return (it != words.end()) ? std::string(*it) : "";
151}

153std::string AppHome::update_param_value_by_index(const std::string &input, size_t index, const std::string &newValue)
154{
155 auto view = input | std::views::split(' ') | std::views::transform(
156 [](auto &&range) {return std::string(range.begin(), range.end());}
157 );
158
159 std::vector<std::string> words{view.begin(), view.end()};
160
161 if (index < words.size())
162 {
163 words[index] = newValue;
164 }
165
166 if (words.empty())
167 {
168 return "";
169 }
170
171 return std::accumulate(
172 std::next(words.begin()),
173 words.end(),
174 words[0],
175 [](std::string a, std::string b)
176 {
177 return std::move(a) + " " + b;
178 }
179 );
180}
