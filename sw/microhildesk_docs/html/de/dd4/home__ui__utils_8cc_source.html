140std::string AppHome::update_param_value_by_index(const std::string_view &input, size_t index, const std::string_view &newValue)
141{
142 auto elements = split_to_vector(input);
143
144 if (elements.size() < cNumOfChannels)
145 {
146 std::string defaultValue = elements.empty() ? std::string(newValue) : elements.back();
147 elements.resize(cNumOfChannels, defaultValue);
148 }
149
150 if (index < elements.size())
151 {
152 elements[index] = std::string(newValue);
153 }
154
155 std::string result;
156 for (size_t i = 0; i < elements.size(); ++i)
157 {
158 result += elements[i] + (i < elements.size() - 1 ? std::string(cSpaceString) : "");
159 }
160
161 return result;
162}
