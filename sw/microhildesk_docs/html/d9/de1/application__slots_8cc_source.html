52void EntryApplication::on_startup()
53{
54 std::cout << "Startup application..." << std::endl;
55 Gtk::Application::on_startup();
56
59 auto menu = Gio::Menu::create();
60 auto submenu_file = Gio::Menu::create();
61 submenu_file->append(fileQuitLabel.data(), fileQuitDetaildAction.data());
62 menu->append_submenu(fileLabel.data(), submenu_file);
63 auto submenu_option = Gio::Menu::create();
64 submenu_option->append(optionSettingsLabel.data(), optionSettingsDetaildAction.data());
65 menu->append_submenu(optionLabel.data(), submenu_option);
66 auto submenu_help = Gio::Menu::create();
67 submenu_help->append(helpDocLabel.data(), helpDocDetaildAction.data());
68 submenu_help->append(helpAboutLabel.data(), helpAboutDetaildAction.data());
69 menu->append_submenu(helpLabel.data(), submenu_help);
70 set_menubar(menu);
71
74 mapping();
75
78 add_window(m_home);
79
82 SettingsSetup setup;
83 setup.m_controlConfig = m_configManager.getControlConfig();
84 setup.m_serialConfig = m_configManager.getSerialConfig();
85 setup.m_logConfig = m_configManager.getLogConfig();
86 m_settings.setSettingsSetup(setup);
87 m_settings.updateUiData();
88
91 m_home.set_visible(true);
92 std::cout << "Startup application done." << std::endl;
93}
