51void EntryApplication::on_startup()
52{
53 std::cout << "Startup application..." << std::endl;
54 Gtk::Application::on_startup();
55
56 // Initialize configuration manager (load files, create defaults if needed)
57 m_configManager->init();
58
59 // Sets application menubar with File, Option and Help menu
60 auto menu = Gio::Menu::create();
61 auto submenu_file = Gio::Menu::create();
62 submenu_file->append(cFileQuitLabel.data(), cFileQuitDetailedAction.data());
63 menu->append_submenu(cFileLabel.data(), submenu_file);
64
65 auto submenu_option = Gio::Menu::create();
66 submenu_option->append(cOptionSettingsLabel.data(), cOptionSettingsDetailedAction.data());
67 menu->append_submenu(cOptionLabel.data(), submenu_option);
68
69 auto submenu_help = Gio::Menu::create();
70 submenu_help->append(cHelpDocLabel.data(), cHelpDocDetailedAction.data());
71 submenu_help->append(cHelpAboutLabel.data(), cHelpAboutDetailedAction.data());
72 menu->append_submenu(cHelpLabel.data(), submenu_help);
73 set_menubar(menu);
74
75 // Maps signal/slots for application views and actions
76 mapping();
77
78 // Sets AppHome as toplevel window, add to the application window
79 add_window(m_home);
80
81 // Updated settings widgets data based on loaded configuration
82 SettingsSetup setup;
83 setup.m_config = m_configManager->getConfig();
84 m_settings.setSettingsSetup(setup);
85 m_settings.updateUiData();
86 m_home.setControlSetup(setup);
87 m_home.updateUiData();
88
89 std::cout << "Startup application done." << std::endl;
90}
