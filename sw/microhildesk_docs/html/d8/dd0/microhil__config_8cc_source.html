56bool MHConfig::load()
57{
60 auto loadedConfig = m_config.load_from_file(m_configFilePath);
61
64 auto validatedConfig = validate();
65
66 if (!loadedConfig || !validatedConfig)
67 {
70 return false;
71 }
72
75 auto logPath = getLogPath();
76 auto logLevel = logLevelStringToInt(getLogLevel());
77 m_logConfig.emit(logPath, logLevel);
78
79 VecUInt serialParams{};
80 serialParams.push_back(getBaudRate());
81 serialParams.push_back(getDataBits());
82 serialParams.push_back(parityUnicodeStringToInt(getParity()));
83 serialParams.push_back(getStopBits());
84
87 auto device = getDevice();
88 m_serialConfig.emit(device, serialParams);
89
90 return true;
91}

122bool MHConfig::checkConfigPath() const
123{
126 std::filesystem::directory_entry homeDirEntry{m_homePath};
127 const auto homeDirExists = homeDirEntry.exists();
128
129 if (!homeDirExists)
130 {
131 if (!std::filesystem::create_directory(m_homePath))
132 {
135 return false;
136 }
137 }
138
142 const auto configFileExists = std::filesystem::exists(m_configFilePath);
143
144 if (!configFileExists)
145 {
146 std::ofstream defaultConfig(m_configFilePath);
147
148 for (int i = 0; i < kConfigLength; i++)
149 {
150 defaultConfig << kConfigSerialDefault[i] << "\n";
151 }
152
153 defaultConfig.close();
154 }
155
156 return true;
157}
