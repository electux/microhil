67bool MicroHILConfig::load()
68{
71 auto loadedConfig = m_config.load_from_file(m_configFilePath);
72
75 auto validatedConfig = validate();
76
77 if(!loadedConfig || !validatedConfig)
78 {
81 return false;
82 }
83
86 m_logConfig.emit(getLogPath(), logLevelStringToInt(getLogLevel()));
87
88 VectorUInt serialParams{};
89 serialParams.push_back(getBaudRate());
90 serialParams.push_back(getDataBits());
91 serialParams.push_back(parityUnicodeStringToInt(getParity()));
92 serialParams.push_back(getStopBits());
93
96 m_serialConfig.emit(getDevice(), serialParams);
97
98 return true;
99}

128bool MicroHILConfig::checkConfigPath()
129{
131 // Check home directory /home/<username>/.microhil/
132 std::filesystem::directory_entry homeDirEntry{m_homePath};
133 auto homeDirExists = homeDirEntry.exists();
134
135 if(!homeDirExists)
136 {
137 if(!std::filesystem::create_directory(m_homePath))
138 {
141 return false;
142 }
143 }
144
146 // Check config file in case of missing generate new with default setup
147 // Configuration file location: /home/<username>/.microhil/config
148 auto configFileExists = std::filesystem::exists(m_configFilePath);
149
150 if(!configFileExists)
151 {
152 std::ofstream defaultConfig(m_configFilePath);
153
154 for (int i = 0; i < kConfigLength; i++)
155 {
156 defaultConfig << kConfigSerialDefault[i] << "\n";
157 }
158
159 defaultConfig.close();
160 }
161
162 return true;
163}
