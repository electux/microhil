62bool MHConfig::load()
63{
66 auto loadedConfig = m_config.load_from_file(m_configFilePath);
67
70 auto validatedConfig = validate();
71
72 if (!loadedConfig || !validatedConfig)
73 {
76 return false;
77 }
78
81 auto logPath = getLogPath();
82 auto logLevel = logLevelStringToInt(getLogLevel());
83 m_logConfig.emit(logPath, logLevel);
84
85 MHVecUInt serialParams{};
86 serialParams.push_back(getBaudRate());
87 serialParams.push_back(getDataBits());
88 serialParams.push_back(parityUnicodeStringToInt(getParity()));
89 serialParams.push_back(getStopBits());
90
93 auto device = getDevice();
94 m_serialConfig.emit(device, serialParams);
95
96 return true;
97}

126bool MHConfig::checkConfigPath() const
127{
129 // Checks home directory /home/<username>/.microhil/
130 std::filesystem::directory_entry homeDirEntry{m_homePath};
131 const auto homeDirExists = homeDirEntry.exists();
132
133 if (!homeDirExists)
134 {
135 if (!std::filesystem::create_directory(m_homePath))
136 {
139 return false;
140 }
141 }
142
144 // Checks config file in case of missing generate new with default setup
145 // Configuration file location: /home/<username>/.microhil/config
146 const auto configFileExists = std::filesystem::exists(m_configFilePath);
147
148 if (!configFileExists)
149 {
150 std::ofstream defaultConfig(m_configFilePath);
151
152 for (int i = 0; i < kConfigLength; i++)
153 {
154 defaultConfig << kConfigSerialDefault[i] << "\n";
155 }
156
157 defaultConfig.close();
158 }
159
160 return true;
161}
