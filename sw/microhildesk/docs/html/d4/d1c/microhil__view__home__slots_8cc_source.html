149void MicroHILViewHome::onChannel0TypeSelected()
150{
151 if(m_enableChannels[0]->get_active() == true)
152 {
153 int controlType = m_selectControlChannels[0]->get_active_row_number();
154
155 switch(controlType)
156 {
157 case static_cast<int>(channelControlType::TOGGLE_BUTTON):
158 toggleModeChannel0();
159 break;
160 case static_cast<int>(channelControlType::TIMER_BUTTON):
161 timerModeChannel0();
162 break;
163 }
164
165 m_channel0ControlType.emit(controlType);
