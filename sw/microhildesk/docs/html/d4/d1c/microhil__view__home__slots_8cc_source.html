21void MicroHILViewHome::onChannelChanged(Channel id)
22{
23 auto ix = static_cast<int>(id);
24 bool status = m_enableChannels[ix]->get_active();
25
26 if(status)
27 {
28 enableChannel(id);
29 }
30 else
31 {
32 disableChannel(id);
33 }
34
35 m_channelEnabled.emit(id, status);
36}

38void MicroHILViewHome::onTypeSelected(Channel id)
39{
40 auto ix = static_cast<int>(id);
41
42 if(m_enableChannels[ix]->get_active() == true)
43 {
44 int controlType = m_selectControlChannels[ix]->get_active_row_number();
45 switch(controlType)
46 {
47 case static_cast<int>(channelControlType::TOGGLE_BUTTON):
48 toggleModeChannel(id);
49 break;
50 case static_cast<int>(channelControlType::TIMER_BUTTON):
51 timerModeChannel(id);
52 break;
53 }
54
55 m_channelControlType.emit(id, controlType);
56 }
57}
