22void MicroHILViewHome::onChannelChanged(Channel id)
23{
24 auto ix = static_cast<int>(id);
25 bool status = m_enableChannels[ix]->get_active();
26
27 if(status)
28 {
29 enableChannel(id);
30 }
31 else
32 {
33 disableChannel(id);
34 }
35
36 m_channelEnabled.emit(id, status);
37}

39void MicroHILViewHome::onTypeSelected(Channel id)
40{
41 auto ix = static_cast<int>(id);
42
43 if(m_enableChannels[ix]->get_active() == true)
44 {
45 int controlType = m_selectControlChannels[ix]->get_active_row_number();
46 switch(controlType)
47 {
48 case static_cast<int>(channelControlType::TOGGLE_BUTTON):
49 toggleModeChannel(id);
50 break;
51 case static_cast<int>(channelControlType::TIMER_BUTTON):
52 timerModeChannel(id);
53 break;
54 }
55
56 m_channelControlType.emit(id, controlType);
57 }
58}
