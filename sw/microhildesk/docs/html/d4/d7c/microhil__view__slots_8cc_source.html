85void MicroHILView::onChannel0TypeSelected()
86{
87 if(m_enableChannels[0]->get_active() == true)
88 {
89 int controlType = m_selectControlChannels[0]->get_active_row_number();
90
91 switch(controlType)
92 {
93 case static_cast<int>(channelControlType::TOGGLE_BUTTON):
94 toggleModeChannel0();
95 break;
96 case static_cast<int>(channelControlType::TIMER_BUTTON):
97 timerModeChannel0();
98 break;
99 }
100
101 m_channel0ControlType.emit(controlType);
102 }
103}
