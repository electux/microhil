116Glib::ustring MicroHILConfig::parityToUnicodeString(int parity)
117{
120 auto parityPrepared = static_cast<ParityConfig>(parity);
121
122 switch(parityPrepared)
123 {
124 case ParityConfig::MICROHIL_NONE_CONFIG:
125 return kConfigParityNone;
126 case ParityConfig::MICROHIL_ODD_CONFIG:
127 return kConfigParityOdd;
128 case ParityConfig::MICROHIL_EVEN_CONFIG:
129 return kConfigParityEven;
130 case ParityConfig::MICROHIL_MARK_CONFIG:
131 return kConfigParityMark;
132 case ParityConfig::MICROHIL_SPACE_CONFIG:
133 return kConfigParitySpace;
134 }
135
136 return kConfigParityNone;
137}

139int MicroHILConfig::parityUnicodeStringToInt(Glib::ustring parity)
140{

152

155 return 3;

157

160 return 4;
161 }
162
