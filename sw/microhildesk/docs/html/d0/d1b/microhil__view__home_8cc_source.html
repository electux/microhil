112MicroHILViewHome::MicroHILViewHome(
113 BaseObjectType* object, Glib::RefPtr<Gtk::Builder> const& ui
114): Gtk::ApplicationWindow(object), m_ui{ui}
115{
117 // Bind channels
118 for (int i = 0; i < kNumberOfChannels; i++)
119 {
120 m_enableChannels.push_back(
121 Glib::RefPtr<Gtk::CheckButton>::cast_dynamic(
122 m_ui->get_object(kCheckButtonsId[i])
123 )
124 );
125 m_selectControlChannels.push_back(
126 Glib::RefPtr<Gtk::ComboBoxText>::cast_dynamic(
127 m_ui->get_object(kComboBoxTextsId[i])
128 )
129 );
130 m_toggleChannels.push_back(
131 Glib::RefPtr<Gtk::ToggleButton>::cast_dynamic(
132 m_ui->get_object(kToggleButtonsId[i])
133 )
134 );
135 m_spinTimerChannels.push_back(
136 Glib::RefPtr<Gtk::SpinButton>::cast_dynamic(
137 m_ui->get_object(kTimerSpinButtonsId[i])
138 )
139 );
140 m_toggleTimerChannels.push_back(
141 Glib::RefPtr<Gtk::ToggleButton>::cast_dynamic(
142 m_ui->get_object(kTimerToggleButtonsId[i])
143 )
144 );
145 m_stautsTimerChannels.push_back(
146 Glib::RefPtr<Gtk::ProgressBar>::cast_dynamic(
147 m_ui->get_object(kTimerProgressBarStatus0Id[i])
148 )
149 );
150 }
151
153 // Map channels (signals and slots)
154 mapping();
155
157 // Disabled all channels by default
158 for (int i = 0; i < kNumberOfChannels; i++)
159 {
160 disableChannel(static_cast<Channel>(i));
161 }
162}
