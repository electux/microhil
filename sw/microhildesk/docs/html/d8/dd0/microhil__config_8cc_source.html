77bool MicroHILConfig::validate()
78{
79 if(!isPreValid())
80 {
81 return false;
82 }
83
84 auto configCheck = (
85 m_configuration.has_key(kConfigSerialSection, kConfigSerialDevice) &&
86 m_configuration.has_key(kConfigSerialSection, kConfigSerialBaudRate) &&
87 m_configuration.has_key(kConfigSerialSection, kConfigSerialDataBits) &&
88 m_configuration.has_key(kConfigSerialSection, kConfigSerialParity) &&
89 m_configuration.has_key(kConfigSerialSection, kConfigSerialStopBits) &&
90 m_configuration.has_key(kConfigLogSection, kConfigLogLevel) &&
91 m_configuration.has_key(kConfigLogSection, kConfigLogFile)
92 );
93
94 if(!configCheck)
95 {
96 return false;
97 }
98
99 return true;
100}

147bool MicroHILConfig::checkConfigPath()
148{
150 // Check home directory /home/<username>/.microhil/
151 std::filesystem::directory_entry homeDirEntry{m_homePath};
152 auto homeDirExists = homeDirEntry.exists();
153
154 if(!homeDirExists)
155 {
156 if(!std::filesystem::create_directory(m_homePath))
157 {
158 return false;
159 }
160 }
161
163 // Check config file in case of missing generate new with default setup
164 // Configuration file location: /home/<username>/.microhil/config
165 auto configFileExists = std::filesystem::exists(m_configFilePath);
166
167 if(!configFileExists)
168 {
169 std::ofstream defaultConfig(m_configFilePath);
170 for (int i = 0; i < kConfigSerialLength; i++)
171 {
172 defaultConfig << kConfigSerialDefault[i] << "\n";
173 }
174 defaultConfig.close();
175 }
176
177 return true;
178}
