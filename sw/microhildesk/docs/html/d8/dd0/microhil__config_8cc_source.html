127bool MicroHILConfig::checkConfigPath()
128{
130 // Check home directory /home/<username>/.microhil/
131 std::filesystem::directory_entry homeDirEntry{m_homePath};
132 auto homeDirExists = homeDirEntry.exists();
133
134 if(!homeDirExists)
135 {
136 if(!std::filesystem::create_directory(m_homePath))
137 {
138 return false;
139 }
140 }
141
143 // Check config file in case of missing generate new with default setup
144 // Configuration file location: /home/<username>/.microhil/config
145 auto configFileExists = std::filesystem::exists(m_configFilePath);
146
147 if(!configFileExists)
148 {
149 std::ofstream defaultConfig(m_configFilePath);
150
151 for (int i = 0; i < kConfigLength; i++)
152 {
153 defaultConfig << kConfigSerialDefault[i] << "\n";
154 }
155
156 defaultConfig.close();
157 }
158
159 return true;
160}
