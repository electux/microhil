120bool MicroHILConfig::validate()
121{
122 if(!isPreValid())
123 {
124 return false;
125 }
126
129 auto configCheck = (
130 m_configuration.has_key(kConfigSerialSection, kConfigSerialDevice) &&
131 m_configuration.has_key(kConfigSerialSection, kConfigSerialBaudRate) &&
132 m_configuration.has_key(kConfigSerialSection, kConfigSerialDataBits) &&
133 m_configuration.has_key(kConfigSerialSection, kConfigSerialParity) &&
134 m_configuration.has_key(kConfigSerialSection, kConfigSerialStopBits) &&
135 m_configuration.has_key(kConfigLogSection, kConfigLogLevel) &&
136 m_configuration.has_key(kConfigLogSection, kConfigLogFile)
137 );
138
139 if(!configCheck)
140 {
141 return false;
142 }
143
144 return true;
145}

192bool MicroHILConfig::checkConfigPath()
193{
195 // Check home directory /home/<username>/.microhil/
196 std::filesystem::directory_entry homeDirEntry{m_homePath};
197 auto homeDirExists = homeDirEntry.exists();
198
199 if(!homeDirExists)
200 {
201 if(!std::filesystem::create_directory(m_homePath))
202 {
203 return false;
204 }
205 }
206
208 // Check config file in case of missing generate new with default setup
209 // Configuration file location: /home/<username>/.microhil/config
210 auto configFileExists = std::filesystem::exists(m_configFilePath);
211
212 if(!configFileExists)
213 {
214 std::ofstream defaultConfig(m_configFilePath);
215 for (int i = 0; i < kConfigSerialLength; i++)
216 {
217 defaultConfig << kConfigSerialDefault[i] << "\n";
218 }
219 defaultConfig.close();
220 }
221
222 return true;
223}
