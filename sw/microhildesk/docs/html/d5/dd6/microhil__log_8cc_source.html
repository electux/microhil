79void MicroHILLog::write(const Glib::ustring message, LogLevel level)
80{
81 const auto currentTime = getCurrentDateTime();
82 const auto prefixLogLevel = toStringLogType(level);
83 Glib::ustring logMessage = currentTime + prefixLogLevel + message;
84
85 if(m_logFile.is_open() && m_fileOpened)
86 {
87 m_logFile << logMessage;
88 }
89}

102Glib::ustring MicroHILLog::getCurrentDateTime() const
103{
104 time_t rawTime;
105 char buffer[80];
106
107 time(&rawTime);
108 const struct tm *timeInfo = localtime(&rawTime);
109 strftime(buffer, 80, "%d-%m-%Y %I:%M:%S", timeInfo);
110
111 return Glib::ustring(buffer);
112}
