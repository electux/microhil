74MicroHILView::MicroHILView(
75 BaseObjectType* object, Glib::RefPtr<Gtk::Builder> const& ui
76): Gtk::ApplicationWindow(object), m_ui{ui}
77{
79 // Bind channel widgets
80 for (int i = 0; i < kNumberOfChannels; i++)
81 {
82 m_enableChannels.push_back(
83 Glib::RefPtr<Gtk::CheckButton>::cast_dynamic(
84 m_ui->get_object(kCheckButtonsId[i])
85 )
86 );
87 m_selectControlChannels.push_back(
88 Glib::RefPtr<Gtk::ComboBoxText>::cast_dynamic(
89 m_ui->get_object(kComboBoxTextsId[i])
90 )
91 );
92 m_toggleChannels.push_back(
93 Glib::RefPtr<Gtk::ToggleButton>::cast_dynamic(
94 m_ui->get_object(kToggleButtonsId[i])
95 )
96 );
97 m_spinTimerChannels.push_back(
98 Glib::RefPtr<Gtk::SpinButton>::cast_dynamic(
99 m_ui->get_object(kTimerSpinButtonsId[i])
100 )
101 );
102 m_toggleTimerChannels.push_back(
103 Glib::RefPtr<Gtk::ToggleButton>::cast_dynamic(
104 m_ui->get_object(kTimerToggleButtonsId[i])
105 )
106 );
107 m_stautsTimerChannels.push_back(
108 Glib::RefPtr<Gtk::ProgressBar>::cast_dynamic(
109 m_ui->get_object(kTimerProgressBarStatus0Id[i])
110 )
111 );
112 }
113
115 // Map channel signals and slots
116 mapping();
117
119 // Disabled all channels by default
120 disableChannel0();
121 disableChannel1();
122 disableChannel2();
123 disableChannel3();
124}
