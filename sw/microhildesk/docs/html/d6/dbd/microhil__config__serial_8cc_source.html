146bool MicroHILConfig::validateSerialSettings()
147{
150 auto configCheck = (
151 m_config.has_key(kConfigSerialSection, kConfigSerialDevice) &&
152 m_config.has_key(kConfigSerialSection, kConfigSerialBaudRate) &&
153 m_config.has_key(kConfigSerialSection, kConfigSerialDataBits) &&
154 m_config.has_key(kConfigSerialSection, kConfigSerialParity) &&
155 m_config.has_key(kConfigSerialSection, kConfigSerialStopBits)
156 );
157
158 if(!configCheck)
159 {
162 return false;
163 }
164
165 return true;
166}
