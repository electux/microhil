28uint16_t microhil_fetch_command(uint8_t *buffer)
29{
30 uint16_t buffer_index = 0;
31
32 while (true)
33 {
34 int byte = getchar_timeout_us(100);
35
36 if (byte != PICO_ERROR_TIMEOUT && buffer_index < MICROHIL_CMD_LEN)
37 {
38 buffer[buffer_index++] = (byte & 0xFF);
39 }
40 else
41 {
42 buffer[buffer_index] = 0;
43 break;
44 }
45 }
46
47 return buffer_index;
48}

54unsigned char *microhil_extract_command_property(unsigned char *cmd)
55{
56 int cmd_len = strlen(cmd);
57 int prefix_len = strlen(MICROHIL_CMD_PREFIX);
58 int len = cmd_len - prefix_len;
59
62 int status = strncmp(cmd, MICROHIL_CMD_PREFIX, prefix_len);
63
64 if (status != 0)
65 {
68 return NULL;
69 }
70
73 unsigned char *cmd_property = (unsigned char *)malloc(len);
74 sprintf(cmd_property, "%s", cmd + prefix_len);
75
78 return cmd_property;
79}

106int microhil_process_channel_on(unsigned char *cmd_property, int index)
107{

145channel_gpio_num microhil_process_command(unsigned char *cmd)
146{
147 unsigned char *cmd_property = microhil_extract_command_property(cmd);
148
149 if (cmd_property == NULL)
150 {
153 printf("%s %s\n", MICROHIL_CHANNEL_ERROR, MICROHIL_EXT_PROP_FAILED);
154 return MICROHIL_CHANNELS_ERROR;
155 }
156
157 if (microhil_process_all_on(cmd_property) == 0)
158 {
159 free(cmd_property);
162 return MICROHIL_ALL_CHANNELS_ON;
163 }
164
165 if (microhil_process_all_off(cmd_property) == 0)
166 {
167 free(cmd_property);
170 return MICROHIL_ALL_CHANNELS_OFF;
171 }
172
173 for (int i = 0; i < MICROHIL_NUMBER_OF_CHANNELS; i++)
174 {
175 if (microhil_process_channel_on(cmd_property, i) == 0)
176 {
177 free(cmd_property);
180 return index_to_gpio(i);
181 }
182
183 if (microhil_process_channel_off(cmd_property, i) == 0)
184 {
185 free(cmd_property);
188 return index_to_gpio(i);
189 }
190 }
191
192 free(cmd_property);
195 printf("%s %s\n", MICROHIL_CHANNEL_ERROR, MICROHIL_NO_SUPPORT);
196 return MICROHIL_CHANNELS_ERROR;
197}
