29bool microhil_init()

31 bool status = false;

35 status = stdio_init_all();
36
37 if (!status)

41 printf("%s %s\n", MICROHIL_CHANNEL_ERROR, MICROHIL_INIT_FAILED);
42 return status;

47 status = stdio_usb_init();

83 status = microhil_init_buzzer();
84

89 printf("%s %s\n", MICROHIL_CHANNEL_ERROR, MICROHIL_INIT_FAILED);

92

96
97 if (!status)

101 printf("%s %s\n", MICROHIL_CHANNEL_ERROR, MICROHIL_INIT_FAILED);
102 return status;
103 }
104

108
111 microhil_digital_write(MICROHIL_BUZZER, true);
112
113 printf("%s %s\n", MICROHIL_CHANNEL_INFO, MICROHIL_INIT_DONE);
114
115 return status;
