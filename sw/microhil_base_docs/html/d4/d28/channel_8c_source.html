131void microhil_channel(uint8_t *args)
132{
133 uint8_t channel_number = args[6];
134 size_t on_len = strlen(MICROHIL_ON_CHANNEL);
135 size_t off_len = strlen(MICROHIL_OFF_CHANNEL);
136 bool on = (memcmp(args + 8, MICROHIL_ON_CHANNEL, on_len) == 0);
137 bool off = (memcmp(args + 8, MICROHIL_OFF_CHANNEL, off_len) == 0);
138
139 if(on)
140 {
141 microhil_relay(to_gpio(channel_number), on);
142 printf("%s channel %c on\n", MICROHIL_CHANNEL_INFO, channel_number);
143 }
144
145 if (off)
146 {
147 microhil_relay(to_gpio(channel_number), off);
148 printf("%s channel %c off\n", MICROHIL_CHANNEL_INFO, channel_number);
149 }
150}

152void microhil_all_channels(uint8_t *args)
153{
154 size_t on_len = strlen(MICROHIL_ON_CHANNEL);
155 size_t off_len = strlen(MICROHIL_OFF_CHANNEL);
156 bool on = (memcmp(args + 10, MICROHIL_ON_CHANNEL, on_len) == 0);
157 bool off = (memcmp(args + 10, MICROHIL_OFF_CHANNEL, off_len) == 0);
158
159 if (on)
160 {
163 microhil_relay_all(true);
164
167 memset(channels, 1, MICROHIL_NUMBER_OF_CHANNELS * sizeof(uint8_t));
168 printf("%s %s\n", MICROHIL_CHANNEL_INFO, MICROHIL_ALL_ON);
169 }
170
171 if (off)
172 {
175 microhil_relay_all(false);
176
179 memset(channels, 0, MICROHIL_NUMBER_OF_CHANNELS * sizeof(uint8_t));
180 printf("%s %s\n", MICROHIL_CHANNEL_INFO, MICROHIL_ALL_OFF);
181 }
