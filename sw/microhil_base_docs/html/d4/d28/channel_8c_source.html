158MICROHIL_PRIVATE void microhil_channel(uint8_t *args)
159{
160 uint8_t channel_number = args[6];
161 size_t on_len = strlen(MICROHIL_ON_CHANNEL);
162 size_t off_len = strlen(MICROHIL_OFF_CHANNEL);
163 bool on = (memcmp(args + 8, MICROHIL_ON_CHANNEL, on_len) == 0);
164 bool off = (memcmp(args + 8, MICROHIL_OFF_CHANNEL, off_len) == 0);
165
166 if(on)
167 {
168 microhil_relay(to_gpio(channel_number), on);
169#ifdef VERBOSE
170 printf("%s channel %c on\n", MICROHIL_VERBOSE, channel_number);
171#endif
172 }
173
174 if (off)
175 {
176 microhil_relay(to_gpio(channel_number), off);
177#ifdef VERBOSE
178 printf("%s channel %c off\n", MICROHIL_VERBOSE, channel_number);
179#endif
180 }
181}

186MICROHIL_PRIVATE void microhil_all_channels(uint8_t *args)
187{
188 size_t on_len = strlen(MICROHIL_ON_CHANNEL);
189 size_t off_len = strlen(MICROHIL_OFF_CHANNEL);
190 bool on = (memcmp(args + 10, MICROHIL_ON_CHANNEL, on_len) == 0);
191 bool off = (memcmp(args + 10, MICROHIL_OFF_CHANNEL, off_len) == 0);
192
193 if (on)
194 {
197 microhil_relay_all(true);
198
201 memset(channels, 1, MICROHIL_NUMBER_OF_CHANNELS * sizeof(uint8_t));
202#ifdef VERBOSE
203 printf("%s %s\n", MICROHIL_VERBOSE, MICROHIL_ALL_ON);
204#endif
205 }
206
207 if (off)
208 {
211 microhil_relay_all(false);
212
215 memset(channels, 0, MICROHIL_NUMBER_OF_CHANNELS * sizeof(uint8_t));
216#ifdef VERBOSE
217 printf("%s %s\n", MICROHIL_VERBOSE, MICROHIL_ALL_OFF);
218#endif
219 }
220
223 microhil_write_pixel(255, 255, 255);
224
227 microhil_write_pwm(PWM_CHAN_A, 80);
228 microhil_delay_ms(100);
229 microhil_write_pwm(PWM_CHAN_A, 0);
230}
