83bool microhil_init_pwm(uint gpio, enum pwm_chan channel)
84{
85 bool status = false;
86
89 gpio_set_function(gpio, GPIO_FUNC_PWM);
90
93 slice_num = pwm_gpio_to_slice_num(gpio);
94 pwm_set_wrap(slice_num, 500);
95 pwm_set_chan_level(slice_num, channel, 1);
96 pwm_set_clkdiv(slice_num, 50);
97 pwm_set_enabled(slice_num, true);
98 status = true;
99
100 return status;
101}
