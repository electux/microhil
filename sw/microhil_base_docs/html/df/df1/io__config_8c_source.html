90bool microhil_init_pwm(uint gpio, enum pwm_chan channel)
91{
92 bool status = false;
93
96 gpio_set_function(gpio, GPIO_FUNC_PWM);
97
100 slice_num = pwm_gpio_to_slice_num(gpio);
101 pwm_set_wrap(slice_num, 500);
102 pwm_set_chan_level(slice_num, channel, 1);
103 pwm_set_clkdiv(slice_num, 50);
104 pwm_set_enabled(slice_num, true);
105 status = true;
106
107 return status;
108}
